Price Calculation Improvements

// PMT (Payment) calculation helper function
function PMT(rate, nper, pv) {
  const pvif = Math.pow(1 + rate, nper);
  const pmt = (rate * pv * pvif) / (pvif - 1);
  return pmt;
}

// Enhanced parameter validation
function validateCalculationParams(params) {
  if (!params.eingangId) throw new Error("Missing eingangId");
  if (params.amount <= 0) throw new Error("Invalid amount");
  if (params.years < 1) throw new Error("Invalid years");
  return true;
}

Feature List Implementation

// German label mapping
const labelToResponseKeyMap = {
  Hersteller: "manufactorer",
  Produkt: "productType",
  // ... more mappings
};

// Dynamic feature list update
function updateFeatureLists(responseData) {
  const features = responseData.features || {};
  Object.entries(features).forEach(([key, value]) => {
    const label = labelToResponseKeyMap[key] || key;
    updateFeatureDisplay(label, value);
  });
}

Map Integration Enhancements

// Enhanced map initialization
function initMap() {
  debugUtils.info("Map", "Initializing Google Map");
  try {
    const mapOptions = {
      zoom: 8,
      center: { lat: 46.8182, lng: 8.2275 },
      mapTypeControl: true,
      streetViewControl: true,
      fullscreenControl: true
    };
    const map = new google.maps.Map(document.getElementById("map"), mapOptions);
    window.map = map;
  } catch (error) {
    debugUtils.error("Map", "Error initializing map", { error });
  }
}

Address Handling Optimization

// Enhanced address input handling
const handleAddressInput = debounce(async function () {
  const address = document.getElementById("addressInput").value.trim();
  debugUtils.info("Address", "Processing input", { address });
  
  if (address.length === 0) {
    debugUtils.warn("Address", "Empty address input");
    return;
  }

  try {
    const possibleAddresses = await fetchDataGet(
      `${possibleAddressEndpoint}?address=${encodeURIComponent(address)}`
    );
    updateDropdown(possibleAddresses);
    resetFormValues();
  } catch (error) {
    debugUtils.error("Address", "Error fetching addresses", { error });
  }
}, 300);
